#include <stdio.h>
#include <stdlib.h>

int main() {
    int i;
    int initial_size = 10;
    int additional_size = 5;
    int total_size;

    // malloc: 10개 문자 메모리 할당
    char *arr = (char *)malloc(initial_size * sizeof(char));
    if (arr == NULL) {
        printf("메모리 할당 실패\n");
        return 1;
    }

    // 초기화: 문자 'A'~'J'
    for (i = 0; i < initial_size; i++) {
        arr[i] = 'A' + i;
    }

    printf("초기 배열: ");
    for (i = 0; i < initial_size; i++) {
        printf("%c ", arr[i]);
    }
    printf("\n");

    // realloc: 5개 추가
    total_size = initial_size + additional_size;
    arr = (char *)realloc(arr, total_size * sizeof(char));
    if (arr == NULL) {
        printf("메모리 재할당 실패\n");
        return 1;
    }

    // 새 메모리 초기화: 'K'~'O'
    for (i = initial_size; i < total_size; i++) {
        arr[i] = 'K' + (i - initial_size);
    }

    // 출력
    printf("재할당 후 배열: ");
    for (i = 0; i < total_size; i++) {
        printf("%c ", arr[i]);
    }
    printf("\n");

    // calloc: 새로운 배열 5개 문자 초기화
    char *new_arr = (char *)calloc(5, sizeof(char));
    if (new_arr == NULL) {
        printf("calloc 메모리 할당 실패\n");
        free(arr);
        return 1;
    }

    // 문자 부여: 'a'~'e'
    for (i = 0; i < 5; i++) {
        new_arr[i] = 'a' + i;
    }

    // 출력
    printf("calloc 배열: ");
    for (i = 0; i < 5; i++) {
        printf("%c ", new_arr[i]);
    }
    printf("\n");


    free(arr);
    free(new_arr);

    return 0;
}
